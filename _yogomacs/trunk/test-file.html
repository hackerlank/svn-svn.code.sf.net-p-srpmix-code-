 44 void screen_window_t::refresh(SDL_Rect *r)
 45 {

 46         int x, y, w, h;
 47         foreground(map_rgb(0x000000));

 48         x = 0;
 49         y = 0;
 50         w = width();
 51         h = _top;

 52         fillrect(x, y, w, h);
 53 
 54         y = height() - _bottom;
 55         h = _bottom;

 56         fillrect(x, y, w, h);
 57 
 58         y = _top;
 59         w = _left;

 60         h = height() - _top - _bottom;
 61         fillrect(x, y, w, h);
 62 
 63         x = width() - _right;

 64         w = _right;
 65         fillrect(x, y, w, h);
 66 }

 87 void dashboard_window_t::mode(dashboard_modes_t m)

 88 {
 89         int vis;
 90         _mode = m;

 91         switch(_mode)
 92         {
 93           case DASHBOARD_GAME:

 94                 vis = 1;
 95                 break;
 96           default:

 97                 vis = 0;
 98                 break;
 99         }
100         wmain->visible(vis);

101         whealth->visible(vis);
102         wtemp->visible(vis);
103         wttemp->visible(vis);
104         wradar->visible(vis);

105         dhigh->visible(vis);
106         dscore->visible(vis);
107         dstage->visible(vis);
108         dships->visible(vis);

109         switch(_mode)
110         {
111           case DASHBOARD_OFF:
112                 break;

113           default:
114                 gengine->invalidate();
115                 gengine->flip();
116                 gengine->invalidate();

117                 gengine->flip();
118                 gengine->invalidate();
119                 break;
120         }

121 }

133 void dashboard_window_t::progress_init(float *progtab)

134 {
135         if(!progtab)
136         {
137                 progress_table = (float *)malloc(MAX_PROGRESS * sizeof(float));

138                 progress_bench = (progress_table != NULL);
139         }
140         else
141         {

142                 progress_table = progtab;
143                 progress_bench = 0;
144         }
145         progress_index = 0;
146 }

147 
148 
149 void dashboard_window_t::progress()
150 {

151         if(progress_table)
152         {
153                 if(progress_bench)
154                 {

155                         progress_table[progress_index++] = (float)SDL_GetTicks();
156                         _percent = 0.0f;
157                 }
158                 else

159                         _percent = progress_table[progress_index++];
160         }
161         else
162                 _percent = 50.0f;

163         invalidate();
164         gengine->flip();
165 }
166 
167 

168 void dashboard_window_t::progress_done()
169 {
170         if(progress_bench)
171         {

172                 int i;
173                 int total = SDL_GetTicks() - (int)progress_table[0];

174                 printf("Progress percentages:\n");
175                 printf("---------------------\n");
176                 for(i = 0; i < progress_index; ++i)

177                         printf("\t%f,\n", (progress_table[i] -
178                                         progress_table[0]) * 100.0f / total);
179                 printf("---------------------\n");
180                 free(progress_table);

181                 progress_bench = 0;
182         }
183         progress_table = NULL;
184 }

187 void dashboard_window_t::render_progress()
188 {

189         SDL_Rect r;
190         int x, y, w, h;

191 
192         x = 0;
193         w = (int)(_percent * 0.01f * width() + 0.5f);
194         if(w < 4)

195                 w = 4;
196         else if(w > width())
197                 w = width();

198 
199         h = 16;
200         y = height() - h;
201 
202         r.x = x;

203         r.y = y;
204         r.w = w;
205         r.h = h;
206 
207         foreground(map_rgb(0x000099));

208         rectangle(x, y, w, h);
209 
210         ++x;
211         ++y;
212         w -= 2;

213         h -= 2;
214         foreground(map_rgb(0x0000cc));
215         rectangle(x, y, w, h);
216 
217         ++x;

218         ++y;
219         w -= 2;
220         h -= 2;
221         foreground(map_rgb(0x0000ff));
222         fillrect(x, y, w, h);

223         
224         r.x = 0;
225         r.y = height() - 40;
226         r.w = width();
227         r.h = 12;

228         foreground(map_rgb(0x000000));
229         fillrect(r.x, r.y, r.w, r.h);
230         if(_msg)
231         {

232                 font(B_NORMAL_FONT);
233                 center(height() - 40, _msg);
234         }
235 }

238 #define NIBBLE_W        4
239 #define NIBBLE_H        4

240 #define NIBBLE_TILES    ((SCREEN_WIDTH/NIBBLE_W+1)*(SCREEN_HEIGHT/NIBBLE_H+1))
241 
242 void dashboard_window_t::nibble(int tool)

243 {
244         int i;
245         int x[NIBBLE_TILES];

246         int y[NIBBLE_TILES];
247 
248         mode(DASHBOARD_OFF);
249 

250         if(tool < 0)
251                 tool = (pubrand.get(20) + SDL_GetTicks()) % 5;
252 
253         /* Clear */

254         for(i = 0; i < NIBBLE_TILES; ++i)
255                 x[i] = y[i] = -1;
256 
257         /* Fill in */

258         int ind = 0;
259         int xx = -NIBBLE_W / 2;
260         int yy = -NIBBLE_H / 2;

261         for(i = 0; i < NIBBLE_TILES; ++i)
262         {
263                 ind = pubrand.get() % NIBBLE_TILES;
264                 if(ind >= NIBBLE_TILES)

265                         ind = 0;
266                 while(x[ind] != -1)
267                         if(++ind >= NIBBLE_TILES)
268                                 ind = 0;

269                 x[ind] = xx;
270                 y[ind] = yy;
271                 xx += NIBBLE_W;
272                 if(xx >= SCREEN_WIDTH + NIBBLE_W / 2)

273                 {
274                         xx = -NIBBLE_W / 2;
275                         yy += NIBBLE_H;
276                 }
277         }

278 
279         /* Clear 8000 tiles/second, until all are done. */
280         foreground(map_rgb(0x000000));
281         ind = 0;

282         int last_index;
283         int t = SDL_GetTicks();
284         while(ind < NIBBLE_TILES)

285         {
286                 int nt = SDL_GetTicks();
287                 int dt = nt - t;

288                 t = nt;
289                 last_index = ind;       /* For double buffer mode */
290                 for(i = 0; i < dt * 8; ++i)

291                 {
292                         if(ind >= NIBBLE_TILES)
293                                 break;
294                         switch (tool)

295                         {
296                           case 0:
297                                 fillrect(x[ind] + NIBBLE_W / 2,
298                                                 y[ind] + NIBBLE_H / 2,

299                                                 NIBBLE_W, NIBBLE_H);
300                                 break;
301                           default:
302                                 sprite(x[ind] - 8 + NIBBLE_W / 2,

303                                                 y[ind] - 8 +
304                                                 NIBBLE_H / 2, B_BRUSHES,
305                                                 tool - 1);
306                                 break;

307                         }
308                         ++ind;
309                 }
310                 gengine->invalidate();

311                 gengine->flip();
312                 if(gengine->doublebuffer())
313                 {
314                         for(i = 0; i < dt * 4; ++i)

315                         {
316                                 if(last_index >= NIBBLE_TILES)
317                                         break;
318                                 switch (tool)

319                                 {
320                                   case 0:
321                                         fillrect(x[last_index] +
322                                                         NIBBLE_W / 2,

323                                                         y[last_index] +
324                                                         NIBBLE_H / 2,
325                                                         NIBBLE_W,
326                                                         NIBBLE_H);
327                                         break;

328                                   default:
329                                         sprite(x[last_index] - 8 +
330                                                         NIBBLE_W / 2,
331                                                         y[last_index] - 8 +

332                                                         NIBBLE_H / 2,
333                                                         B_BRUSHES,
334                                                         tool - 1);
335                                         break;

336                                 }
337                                 ++last_index;
338                         }
339                 }
340         }

341         mode(DASHBOARD_BLACK);
342         gengine->flip();
343 }

346 void dashboard_window_t::refresh(SDL_Rect *r)
347 {
348         switch(_mode)

349         {
350           case DASHBOARD_OFF:
351                 break;
352           case DASHBOARD_BLACK:

353                 background(map_rgb(0x000000));
354                 clear();
355                 break;
356           case DASHBOARD_GAME:

357                 sprite(0, 0, B_SCREEN, 0);
358                 break;
359           case DASHBOARD_LOADING:
360                 background(map_rgb(0x000000));

361                 clear();
362                 sprite((width() - 250) / 2, 55, B_LOADING, 0);
363                 render_progress();
364                 break;

365         }
366 }

373 bargraph_t::bargraph_t()

374 {
375         _value = 0.0f;
376         _redmax = 1;
377         _y = -1000;
378         _enabled = 1;

379 }
380 
381 
382 void bargraph_t::value(float val)

383 {
384         if(val < 0.0f)
385                 _value = 0.0f;
386         else

387                 _value = val;
388         if(!_enabled)
389                 return;
390         int y = (int)((height() - 2) * _value);

391         if(y != _y)
392                 invalidate();
393 }
394 
395 

396 void bargraph_t::enable(int ena)
397 {
398         _enabled = ena;

399         invalidate();
400 }
401 
402 
403 void bargraph_t::refresh(SDL_Rect *r)

404 {
405         if(!_enabled)
406         {
407                 clear();

408                 return;
409         }
410         float v = _value;

411         int red, green, blue;
412         if(v > 1.0f)

413         {
414                 blue = 50 + (v - 1.0f) * 512;
415                 if(_redmax)
416                 {

417                         red = 255;
418                         green = blue / 2;
419                 }
420                 else

421                 {
422                         green = 255;
423                         red = blue / 2;
424                 }
425                 v = 1.0f;

426         }
427         else
428         {
429                 if(_redmax)

430                 {
431                         red = (int)(v * 300.0);
432                         green = (int)((1.0 - v) * 400.0);
433                 }

434                 else
435                 {
436                         red = (int)((1.0 - v) * 300.0);
437                         green = (int)(v * 400.0);

438                 }
439                 blue = 50;
440         }
441         if(green > 180)

442                 green = 180;
443         if(red > 230)
444                 red = 230;
445         if(blue > 255)

446                 blue = 255;
447         _y = (int)((height() - 2) * (1.0f - v));
448         foreground(bgcolor);
449         fillrect(0, 0, width(), height());

450         foreground(map_rgb(red, green, blue));
451         fillrect(1, _y + 1, width() - 2, height() - _y - 2);
452 }
